```cpp
import struct
import time
import serial

# Orange Pi UART0
SERIAL_PORT = "/dev/ttyS0"
BAUDRATE = 230400

# 帧头帧尾（用于数据帧校验）
FRAME_HEADER = b"\xFF"
FRAME_FOOTER = b"\xFE"

# 全局变量（用于接收解析和状态记录）
task_id = 0
sub_task_id = 0
rx_frame_start = False
rx_frame_data = bytearray()


def float_to_bytes(value):
    """将浮点数转为4字节（小端序）。"""
    return struct.pack("<f", float(value))


def build_frame(data):
    """构建完整数据帧（帧头+数据+帧尾）。"""
    return FRAME_HEADER + data + FRAME_FOOTER


def send_data(uart, task, x=None, y=None, status_1=None, status_2=None):
    """通过串口发送数据，支持不同任务类型的数据格式。"""
    if task == 0x01:
        """发送坐标数据"""
        if x is None or y is None:
            print("x or y is None")
            return

        data = bytes([task]) + float_to_bytes(x) + float_to_bytes(y)
        frame = build_frame(data)
        uart.write(frame)
        uart.flush()
        print(f"Send：task ID=0x{task:02X}, x={x}, y={y}, frame={frame.hex(' ')}")
        return

    if task == 0x02:
        """发送目标状态"""
        if status_1 is None or status_2 is None:
            print("status_1 or status_2 is None")
            return

        board_state = int(status_1) & 0xFF
        piece_state = int(status_2) & 0xFF
        data = bytes([task, board_state, piece_state])
        frame = build_frame(data)
        uart.write(frame)
        uart.flush()
        print(
            "Send："
            f"task ID=0x{task:02X}, "
            f"status_1=0b{board_state:08b}, "
            f"status_2=0b{piece_state:08b}, "
            f"frame={frame.hex(' ')}"
        )
        return

    print(f"unknown ID：0x{task:02X}")


def handle_frame(frame_data):
    """解析一帧完整数据。"""
    global task_id, sub_task_id

    if len(frame_data) < 4:
        """数据太短"""
        print(f"invalid short frames：{frame_data.hex(' ')}")
        return

    task_id = frame_data[1]
    sub_task_id = frame_data[2]
    print(f"Receive：Task ID=0x{task_id:02X}, sub task ID=0x{sub_task_id:02X}, frame={frame_data.hex(' ')}")

    if task_id == 0x01:
        print(f"execute task：sub task ID=0x{sub_task_id:02X}")
    elif task_id == 0x02:
        print(f"execute task：子任务ID=0x{sub_task_id:02X}")
    elif task_id == 0x03:
        print(f"reset task：子任务ID=0x{sub_task_id:02X}")
        if sub_task_id == 0x02:
            print("confirm reset")
    else:
        print(f"Unknown ID：0x{task_id:02X}")


def receive_and_unpack(uart):
    """接收并解析串口数据帧。"""
    global rx_frame_start, rx_frame_data

    while uart.in_waiting:
        byte = uart.read(1)
        if not byte:
            break

        byte_val = byte[0]
        
        # 确认帧头
        if byte_val == 0xFF: 
            rx_frame_start = True
            rx_frame_data = bytearray([byte_val])
            continue

        # 确认帧尾
        if rx_frame_start and byte_val == 0xFE:
            rx_frame_data.append(byte_val)
            handle_frame(rx_frame_data)
            rx_frame_start = False
            rx_frame_data = bytearray()
            continue

        if rx_frame_start:
            rx_frame_data.append(byte_val)


def open_uart():
    """初始化串口"""
    return serial.Serial(
        port=SERIAL_PORT,
        baudrate=BAUDRATE,
        bytesize=serial.EIGHTBITS,
        parity=serial.PARITY_NONE,
        stopbits=serial.STOPBITS_ONE,
        timeout=0,
    )


def main():
    uart = open_uart()
    last_send_time = 0

    print(f"UART opened: {SERIAL_PORT}, baudrate={BAUDRATE}")

    try:
        while True:
            receive_and_unpack(uart)

            now = time.monotonic()
            if now - last_send_time >= 5:
                send_data(uart, 0x02, x=100.5, y=200.3)
                send_data(uart, 0x03, status_1=0b101010, status_2=0b010101)
                last_send_time = now

            time.sleep(0.02)

    except KeyboardInterrupt:
        print("用户终止程序")
    finally:
        uart.close()
        print("串口关闭")


if __name__ == "__main__":
    main()
import struct
import time
import serial


# Orange Pi UART0
SERIAL_PORT = "/dev/ttyS0"
BAUDRATE = 230400

# 数据包格式：[帧头: 0xAA 0x55] + [长度: 1字节] + [数据内容：Payload] + [CRC校验: 2字节]
FRAME_HEADER = b"\xAA\x55"

# 全局变量（用于接收解析和状态记录）
task_id = 0
sub_task_id = 0
rx_buffer = bytearray()


def crc16_modbus(data):
    """计算CRC16。"""
    crc = 0xFFFF
    for byte in data:
        crc ^= byte
        for _ in range(8):
            if crc & 0x0001:
                crc = (crc >> 1) ^ 0xA001
            else:
                crc >>= 1
    return crc


def float_to_bytes(value):
    """将浮点数转为4字节（小端序）。"""
    return struct.pack("<f", float(value))


def build_frame(payload):
    """构建完整数据帧。CRC校验范围为长度字节+Payload。"""
    if len(payload) > 255:
        raise ValueError("payload length must be <= 255 bytes")

    length = bytes([len(payload)])
    body = length + payload
    crc = crc16_modbus(body).to_bytes(2, byteorder="little")
    return FRAME_HEADER + body + crc


def send_data(uart, task, x=None, y=None, status_1=None, status_2=None):
    """通过串口发送数据，支持不同任务类型的数据格式。"""
    if task == 0x01:
        if x is None or y is None:
            print("x or y is None")
            return

        payload = bytes([task]) + float_to_bytes(x) + float_to_bytes(y)
        frame = build_frame(payload)
        uart.write(frame)
        uart.flush()
        print(f"Send: task ID=0x{task:02X}, x={x}, y={y}, frame={frame.hex(' ')}")
        return

    if task == 0x02:
        if status_1 is None or status_2 is None:
            print("status_1 or status_2 is None")
            return

        value_1 = int(status_1) & 0xFF
        value_2 = int(status_2) & 0xFF
        payload = bytes([task, value_1, value_2])
        frame = build_frame(payload)
        uart.write(frame)
        uart.flush()
        print(
            "Send: "
            f"task ID=0x{task:02X}, "
            f"status=[0b{value_1:08b}, 0b{value_2:08b}], "
            f"frame={frame.hex(' ')}"
        )
        return

    print(f"Unknown task ID: 0x{task:02X}")


def handle_payload(payload, frame):
    """解析一帧完整Payload。"""
    global task_id, sub_task_id

    if len(payload) < 2:
        print(f"Invalid short payload: {payload.hex(' ')}, frame={frame.hex(' ')}")
        return

    task_id = payload[0]
    sub_task_id = payload[1]
    print(f"Receive: task ID=0x{task_id:02X}, sub task ID=0x{sub_task_id:02X}, frame={frame.hex(' ')}")

    if task_id == 0x01:
        print(f"Execute task: sub task ID=0x{sub_task_id:02X}")
    elif task_id == 0x02:
        print(f"Execute task: sub task ID=0x{sub_task_id:02X}")
    elif task_id == 0x03:
        print(f"Reset task: sub task ID=0x{sub_task_id:02X}")
        if sub_task_id == 0x02:
            print("Confirm reset")
    else:
        print(f"Unknown task ID: 0x{task_id:02X}")


def parse_rx_buffer():
    """从接收缓冲区中解析所有完整数据帧。"""
    global rx_buffer

    while True:
        header_index = rx_buffer.find(FRAME_HEADER)
        if header_index < 0:
            rx_buffer.clear()
            return

        if header_index > 0:
            del rx_buffer[:header_index]

        if len(rx_buffer) < 5:
            return

        payload_len = rx_buffer[2]
        frame_len = 2 + 1 + payload_len + 2
        if len(rx_buffer) < frame_len:
            return

        frame = bytes(rx_buffer[:frame_len])
        del rx_buffer[:frame_len]

        body = frame[2:-2]
        received_crc = int.from_bytes(frame[-2:], byteorder="little")
        calculated_crc = crc16_modbus(body)
        if received_crc != calculated_crc:
            print(
                "CRC error: "
                f"recv=0x{received_crc:04X}, "
                f"calc=0x{calculated_crc:04X}, "
                f"frame={frame.hex(' ')}"
            )
            continue

        payload = frame[3:-2]
        handle_payload(payload, frame)


def receive_and_unpack(uart):
    """接收并解析串口数据帧。"""
    global rx_buffer

    waiting = uart.in_waiting
    if waiting <= 0:
        return

    data = uart.read(waiting)
    if not data:
        return

    rx_buffer.extend(data)
    parse_rx_buffer()


def open_uart():
    """初始化Linux串口，替代K230 machine.UART/FPIOA。"""
    return serial.Serial(
        port=SERIAL_PORT,
        baudrate=BAUDRATE,
        bytesize=serial.EIGHTBITS,
        parity=serial.PARITY_NONE,
        stopbits=serial.STOPBITS_ONE,
        timeout=0,
    )


def main():
    uart = open_uart()
    last_send_time = 0

    print(f"UART opened: {SERIAL_PORT}, baudrate={BAUDRATE}")

    try:
        while True:
            receive_and_unpack(uart)

            now = time.monotonic()
            if now - last_send_time >= 5:
                send_data(uart, 0x01, x=100.5, y=200.3)
                send_data(uart, 0x02, status_1=0b101010, status_2=0b010101)
                last_send_time = now

            time.sleep(0.02)

    except KeyboardInterrupt:
        print("User stopped")
    finally:
        uart.close()
        print("UART closed")


if __name__ == "__main__":
    main()
```
